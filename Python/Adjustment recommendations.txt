import re
from decimal import Decimal, InvalidOperation

def normalize_loan_id(value):
    """
    Canonical loan-id normalizer for IRP + CREFC matching.

    Handles:
    - Excel numeric reads: 304810038.0 -> 304810038
    - whitespace / NBSP / quotes
    - commas in numeric strings
    - scientific notation
    - punctuation cleanup for mixed IDs

    Preserves leading zeros for true digit strings like '030000001'.
    """

    if value is None:
        return ""

    s = str(value)

    # normalize whitespace
    s = s.replace("\u00A0", " ")
    s = s.strip().strip("'").strip('"')

    if not s:
        return ""

    s = re.sub(r"\s+", "", s)

    # remove commas
    s_no_commas = s.replace(",", "")

    # preserve leading zero digit strings
    if re.fullmatch(r"\d+", s_no_commas):
        return s_no_commas

    # collapse integer-like decimals safely
    try:
        d = Decimal(s_no_commas)
        if d == d.to_integral_value():
            return str(d.quantize(Decimal("1")))
    except (InvalidOperation, ValueError):
        pass

    # fallback for alphanumeric IDs
    alnum = re.sub(r"[^A-Za-z0-9]", "", s).upper()

    if alnum:
        return alnum

    return s.upper()

def loan_id_variants(value):
    """
    Generate multiple matching keys for a loan ID.
    This helps when files use slightly different formats.
    """

    base = normalize_loan_id(value)

    if not base:
        return []

    variants = {base}

    # numeric variants
    if base.isdigit():

        variants.add(base.lstrip("0") or "0")

        try:
            variants.add(str(int(base)))
        except Exception:
            pass

    # raw alphanumeric cleanup
    raw = str(value).strip()
    raw_alnum = re.sub(r"[^A-Za-z0-9]", "", raw).upper()

    if raw_alnum:
        variants.add(raw_alnum)

    return [v for v in variants if v]

def get_loan_ids(irp_data, trans_id):

    result = {}

    for i, row in enumerate(irp_data[1:], start=1):

        if str(row[cfg.IRP_COL_TRANS_ID] or "").strip() != trans_id:
            continue

        for col in (cfg.IRP_COL_LOAN_ID, cfg.IRP_COL_LOAN_NUM):

            raw = row[col]

            for lid in loan_id_variants(raw):

                result.setdefault(lid, i)

    return result
